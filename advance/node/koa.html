2、路由模块化 ​

app.js

javascript
const Koa = require('koa')
const router = require('./routes')

const app = new Koa()

app.use(router.routes(), router.allowedMethods()).listen(3000)

routes.js

javascript
const Router = require('koa-router')

const userRouter = require('./usre.js')
const listRouter = require('./list.js')

const router = new Router()

//统一添加前缀
router.prefix('/api')
//注册路由中间件
router.use('/user', userRouter.routes(), userRouter.allowedMethods())
router.use('/list', userRouter.routes(), listRouter.allowedMethods())
//路由重定向
router.redirect('/', '/list')

module.export = router

user.js

javascript
const Router = require('koa-router')
const router = new Router()

router
  .post('/user', (ctx, next) => {
    ctx.body = {
      ok: 1,
      msg: 'add user success'
    }
  })
  .get('/user', (ctx, next) => {
    ctx.body = ['111', '222']
  })
  .put('/user/:id', (ctx, next) => {
    ctx.body = {
      ok: 1,
      msg: 'put user success'
    }
  })
  .del('/user/:id', (ctx, next) => {
    ctx.body = {
      ok: 1,
      msg: 'del user success'
    }
  })

module.export = router

5、身份验证 ​

cookie&session ​

javascript
const session = require('koa-session-minimal')

app.use(
  session({
    key: 'koency',
    cookie: {
      maxAge: 3600
    }
  })
)

app.use(async (ctx, next) => {
  if (ctx.url.includes('login')) {
    await next()
    return
  }
  if (ctx.session.user) {
    ctx.session.date = Date.now()
    await next()
  } else {
    ctx.redirect('/login')
  }
})

//login.html
router.post('/login', (ctx) => {
  const { username, password } = ctx.request.body
  if (username === 'koency' && password === '123abc') {
    ctx.body = {
      ok: 1
    }
  } else {
    ctx.body = {
      ok: 0
    }
  }
})
